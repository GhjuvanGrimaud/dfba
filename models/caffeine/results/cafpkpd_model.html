def plot_balance(s):
    """ Plot the balance of the amount. """
    plt.plot(s['time'], s['D_caf'], color='black', label='D')
    plt.plot(s['time'], s['Abody_caf'], color='blue', label='Abody')
    plt.plot(s['time'], s['D_caf'] + s['Abody_caf'], color='red', label='D+Abody')
    plt.plot(s['time'], s['DCL_caf'], color='gray', label='DCL')
    plt.plot(s['time'], s['D_caf'] + s['Abody_caf'] + s['DCL_caf'], label='D+Abody+DCL', color='black')

HLM_values = [1, 1.5, 2]
results = []
for hlm in HLM_values:
    r.resetAll()
    r.setValue('HLM_CLint_caf', hlm)  # oral dose in [mg]
    r.reset(SelectionRecord.INITIAL_GLOBAL_PARAMETER)
    r.setValue('init(PODOSE_caf)', 300)  # IV dose in [mg]
    r.setValue('init(IVDOSE_caf)', 0)  # IV dose in [mg]
    s = r.simulate(0,15,1001)
    results.append(s)
    
for k, hlm in enumerate(HLM_values):
